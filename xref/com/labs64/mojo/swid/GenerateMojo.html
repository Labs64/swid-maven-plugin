37  /**
38   * A mojo that generates a SWID tag from a given POM.
39   * 
40   * @see <a href="http://l64.cc/swid">SoftWare IDentification (SWID) Tags Generator</a>

43  public class GenerateMojo extends AbstractSwidMojo {

186     public void execute() throws MojoExecutionException {
187         getLog().debug("Generate SWID Tag...");
188 
189         // prepare mandatory elements
190         ArtifactVersion artifactVersion = getArtifactVersion();
191         prepareMandatoryElements();
192 
193         // prepare SWID Tag processor
194         SwidProcessor processor = new DefaultSwidProcessor();
195         ((DefaultSwidProcessor) processor).setEntitlementRequiredIndicator(entitlement_required)
196                 .setProductTitle(product_title)
197                 .setProductVersion(product_version,
198                         artifactVersion.getMajorVersion(),
199                         artifactVersion.getMinorVersion(),
200                         artifactVersion.getIncrementalVersion(),
201                         artifactVersion.getBuildNumber())
202                 .setSoftwareCreator(software_creator.getName(), software_creator.getRegid())
203                 .setSoftwareLicensor(software_licensor.getName(), software_licensor.getRegid())
204                 .setSoftwareId(software_id.getUnique_id(), software_id.getTag_creator_regid())
205                 .setTagCreator(tag_creator.getName(), tag_creator.getRegid());
206 
207         // create builder and pass processor as build param
208         SwidBuilder builder = new SwidBuilder();
209         SoftwareIdentificationTagComplexType swidTag = builder.build(processor);
210 
211         // output resulting object
212         final String fileName = SwidUtils.generateSwidFileName(
213                 software_creator.getRegid(),
214                 software_id.getUnique_id(),
215                 product_version,
216                 extension);
217         if (!outputDirectory.exists()) {
218             if (!outputDirectory.mkdirs()) {
219                 throw new MojoExecutionException("Cannot create directory '" + outputDirectory.toString() + "'");
220             }
221         }
222         File swidFile = new File(outputDirectory, fileName);
223         SwidWriter writer = new SwidWriter();
224         writer.write(swidTag, swidFile);
225     }

227     private ArtifactVersion getArtifactVersion() {
228         if (ArtifactUtils.isSnapshot(product_version)) {
229             product_version = StringUtils.substring(product_version, 0, product_version.length()
230                     - Artifact.SNAPSHOT_VERSION.length() - 1);
231         }
232         return new DefaultArtifactVersion(product_version);
233     }
234 
235     private String getDomainDate() {
236         if (StringUtils.isBlank(domain_creation_date)) {
237             domain_creation_date = SwidUtils.generateDomainDate(new Date());
238         }
239         return domain_creation_date;
240     }
241 
242     private RegId getDefaultRegId() {
243         final RegId regid = new RegId();
244 
245         regid.setName(getProject().getOrganization() == null ?
246                 getProject().getGroupId() : getProject().getOrganization().getName());
247 
248         final String url = getProject().getOrganization() == null ?
249                 getProject().getUrl() : getProject().getOrganization().getUrl();
250         final String reverseDomainName = StringUtils.isBlank(url) ?
251                 getProject().getGroupId() : SwidUtils.revertDomainName(url);
252         regid.setRegid(SwidUtils.generateRegId(getDomainDate(), reverseDomainName));
253 
254         return regid;
255     }
256 
257     private void prepareMandatoryElements() {
258         final RegId defaultRegId = getDefaultRegId();
259 
260         // software_creator
261         if (software_creator == null) {
262             software_creator = new RegId();
263         }
264         if (StringUtils.isBlank(software_creator.getName())) {
265             software_creator.setName(defaultRegId.getName());
266         }
267         if (StringUtils.isBlank(software_creator.getRegid())) {
268             software_creator.setRegid(defaultRegId.getRegid());
269         }
270 
271         // software_licensor
272         if (software_licensor == null) {
273             software_licensor = new RegId();
274         }
275         if (StringUtils.isBlank(software_licensor.getName())) {
276             software_licensor.setName(defaultRegId.getName());
277         }
278         if (StringUtils.isBlank(software_licensor.getRegid())) {
279             software_licensor.setRegid(defaultRegId.getRegid());
280         }
281 
282         // tag_creator
283         if (tag_creator == null) {
284             tag_creator = new RegId();
285         }
286         if (StringUtils.isBlank(tag_creator.getName())) {
287             tag_creator.setName(defaultRegId.getName());
288         }
289         if (StringUtils.isBlank(tag_creator.getRegid())) {
290             tag_creator.setRegid(defaultRegId.getRegid());
291         }
292 
293         // software_id
294         if (software_id == null) {
295             software_id = new RegId();
296         }
297         if (StringUtils.isBlank(software_id.getUnique_id())) {
298             software_id.setUnique_id(getProject().getArtifactId());
299         }
300         if (StringUtils.isBlank(software_id.getTag_creator_regid())) {
301             software_id.setTag_creator_regid(tag_creator.getRegid());
302         }
303     }
