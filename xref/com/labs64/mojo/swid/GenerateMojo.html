14  package com.labs64.mojo.swid;
15  
16  import java.io.File;
17  
18  import org.apache.commons.lang3.StringUtils;
19  import org.apache.maven.artifact.Artifact;
20  import org.apache.maven.artifact.ArtifactUtils;
21  import org.apache.maven.artifact.versioning.ArtifactVersion;
22  import org.apache.maven.artifact.versioning.DefaultArtifactVersion;
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.plugins.annotations.LifecyclePhase;
25  import org.apache.maven.plugins.annotations.Mojo;
26  import org.apache.maven.plugins.annotations.Parameter;
27  import org.iso.standards.iso._19770.__2._2009.schema.SoftwareIdentificationTagComplexType;
28  
29  import com.labs64.mojo.swid.configuration.RegId;
30  import com.labs64.utils.swid.SwidBuilder;
31  import com.labs64.utils.swid.io.SwidWriter;
32  import com.labs64.utils.swid.processor.DefaultSwidProcessor;
33  import com.labs64.utils.swid.processor.SwidProcessor;
34  import com.labs64.utils.swid.support.SwidUtils;

36  /**
37   * A mojo that generates a SWID tag from a given POM.
38   * 
39   * @see <a href="http://l64.cc/swid">SoftWare IDentification (SWID) Tags Generator</a>
40   */
41  @Mojo(name = "generate", defaultPhase = LifecyclePhase.PROCESS_RESOURCES)
42  public class GenerateMojo extends AbstractSwidMojo {
43  
44      /**
45       * Specifies the destination directory where the generated SWID tags files will be saved.
46       * 
47       * @since 1.0.0
48       */
49      @Parameter(property = "swid.outputDirectory", required = true, defaultValue = "${project.build.directory}/swid")
50      private File outputDirectory;
51  
52      /**
53       * Specifies the encoding of the generated SWID tags files. If not specified, the encoding value will be UTF-8.
54       * 
55       * @since 1.0.0
56       */
57      @Parameter(property = "swid.encoding", required = false, defaultValue = "${project.build.sourceEncoding}")
58      private String encoding;
59  
60      /**
61       * Specifies the extension of the generated SWID tags files.
62       * 
63       * @since 1.0.0
64       */
65      @Parameter(property = "swid.extension", required = false, defaultValue = "swidtag")
66      private String extension;
67  
68      /**
69       * Specifies if an entitlement is required to reconcile this product.
70       * 
71       * @since 1.0.0
72       */
73      @Parameter(property = "swid.entitlement_required", required = false, defaultValue = "false")
74      private Boolean entitlement_required;
75  
76      /**
77       * Specifies product title.
78       * 
79       * @since 1.0.0
80       */
81      @Parameter(property = "swid.product_title", required = false, defaultValue = "${project.name}")
82      private String product_title;
83  
84      /**
85       * Specifies software creator attributes. <br/>
86       * <b>Default values:</b>
87       * 
88       * <pre>
89       * &lt;software_creator&gt;
90       *     &lt;name&gt;${project.organization.name}&lt;/name&gt;
91       *     &lt;regid&gt;${project.organization.url}&lt;/regid&gt;
92       * &lt;/software_creator&gt;
93       * </pre>
94       * 
95       * @since 1.0.0
96       */
97      @Parameter(property = "swid.software_creator", required = false)
98      private RegId software_creator;
99  
100     /**
101      * Specifies software licensor attributes. <br/>
102      * <b>Default values:</b>
103      * 
104      * <pre>
105      * &lt;software_licensor&gt;
106      *     &lt;name&gt;${project.organization.name}&lt;/name&gt;
107      *     &lt;regid&gt;${project.organization.url}&lt;/regid&gt;
108      * &lt;/software_licensor&gt;
109      * </pre>
110      * 
111      * @since 1.0.0
112      */
113     @Parameter(property = "swid.software_licensor", required = false)
114     private RegId software_licensor;
115 
116     /**
117      * Specifies software identifier attributes. <br/>
118      * <b>Default values:</b>
119      * 
120      * <pre>
121      * &lt;software_id&gt;
122      *     &lt;unique_id&gt;${project.organization.name}&lt;/unique_id&gt;
123      *     &lt;tag_creator_regid&gt;${project.organization.url}&lt;/tag_creator_regid&gt;
124      * &lt;/software_id&gt;
125      * </pre>
126      * 
127      * @since 1.0.0
128      */
129     @Parameter(property = "swid.software_id", required = false)
130     private RegId software_id;
131 
132     /**
133      * Specifies tag creator attributes. <br/>
134      * <b>Default values:</b>
135      * 
136      * <pre>
137      * &lt;tag_creator&gt;
138      *     &lt;name&gt;${project.organization.name}&lt;/name&gt;
139      *     &lt;regid&gt;${project.organization.url}&lt;/regid&gt;
140      * &lt;/tag_creator&gt;
141      * </pre>
142      * 
143      * @since 1.0.0
144      */
145     @Parameter(property = "swid.tag_creator", required = false)
146     private RegId tag_creator;
147 
148     public void execute() throws MojoExecutionException {
149         getLog().debug("Generate SWID Tag...");
150 
151         File dir = outputDirectory;
152         if (!dir.exists()) {
153             dir.mkdirs();
154         }
155 
156         final String regId = SwidUtils.generateRegId("2010-04", "com.labs64");
157         // Parse a version String and add the components to a properties object.
158         String projectVersion = getProject().getVersion();
159         if (ArtifactUtils.isSnapshot(projectVersion)) {
160             projectVersion = StringUtils.substring(projectVersion, 0, projectVersion.length()
161                     - Artifact.SNAPSHOT_VERSION.length() - 1);
162         }
163         ArtifactVersion artifactVersion = new DefaultArtifactVersion(projectVersion);
164 
165         final String fileName = SwidUtils.generateSwidFileName(regId,
166                 getProject().getArtifactId(),
167                 projectVersion,
168                 extension);
169         File touch = new File(dir, fileName);
170 
171         // prepare SWID Tag processor
172         SwidProcessor processor = new DefaultSwidProcessor();
173         ((DefaultSwidProcessor) processor).setEntitlementRequiredIndicator(entitlement_required)
174                 .setProductTitle(product_title)
175                 // TODO
176                 .setProductVersion(projectVersion,
177                         artifactVersion.getMajorVersion(),
178                         artifactVersion.getMinorVersion(),
179                         artifactVersion.getIncrementalVersion(),
180                         artifactVersion.getBuildNumber())
181                 .setSoftwareCreator(software_creator.getName(), software_creator.getRegid())
182                 .setSoftwareLicensor(software_licensor.getName(), software_licensor.getRegid())
183                 .setSoftwareId(software_id.getUnique_id(), software_id.getTag_creator_regid())
184                 .setTagCreator(tag_creator.getName(), tag_creator.getRegid());
185 
186         // create builder and pass processor as build param
187         SwidBuilder builder = new SwidBuilder();
188         SoftwareIdentificationTagComplexType swidTag = builder.build(processor);
189 
190         // output resulting object
191         SwidWriter writer = new SwidWriter();
192         writer.write(swidTag, touch);
193     }
194 
195 }
